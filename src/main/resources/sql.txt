      CREATE TABLE `catalog_db`.`basket_order` (
        `id_basket` INT NOT NULL AUTO_INCREMENT,
        `id_user` INT NOT NULL,
        `id_product` INT NOT NULL,
        PRIMARY KEY (`id_basket`),
        INDEX `id_user_idx` (`id_user` ASC) VISIBLE,
        INDEX `id_product_idx` (`id_product` ASC) VISIBLE,
        CONSTRAINT `id_user`
          FOREIGN KEY (`id_user`)
          REFERENCES `catalog_db`.`users` (`id_users`)
          ON DELETE NO ACTION
          ON UPDATE NO ACTION,
        CONSTRAINT `id_product`
          FOREIGN KEY (`id_product`)
          REFERENCES `catalog_db`.`products` (`id_products`)
          ON DELETE NO ACTION
          ON UPDATE NO ACTION);
